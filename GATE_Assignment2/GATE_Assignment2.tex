\section*{Problem}
(GATE EC-2006 Q.14) Let $x(t) \leftrightarrow X(f)$ be Fourier Transform pair. The Fourier Transform of the signal $x(5t - 3)$ in terms of $X(f)$ is given as
\begin{enumerate}[label={(\Alph*)}]
    \item $\frac{1}{5}e^{-\frac{j6\pi f}{5}}X\brak{\frac{f}{5}}$
    \item $\frac{1}{5}e^{\frac{j6\pi f}{5}}X\brak{\frac{f}{5}}$
    \item $\frac{1}{5}e^{j6\pi f}X\brak{\frac{f}{5}}$
    \item $\frac{1}{5}e^{j6\pi f}X\brak{\frac{f}{5}}$
\end{enumerate}
\section*{Solution}
By the time scaling property of Fourier transform,
\begin{align}
    x(\alpha t)&\fourier{\frac{1}{|\alpha|}}X\brak{\frac{f}{|\alpha|}}\\
    x(5t)&\fourier{\frac{1}{5}}X\brak{\frac{f}{5}}
\end{align}
By the time shifting property of Fourier transform,
\begin{align}
    x(t - t_0)&\fourier{e^{-j2\pi ft_0}X(f)}\\
    x\brak{5\brak{t-\frac{3}{5}}}&\fourier{\frac{1}{5}e^{-\frac{j6\pi f}{5}}X\brak{\frac{f}{5}}}
\end{align}
Let $x(t) = \rect{\brak{t}}$ and $\sinc(t) = \frac{\sin(t)}{t}$
\begin{align}
    X(f) = \sinc\brak{f}
\end{align}
For $x(t) = \rect{\brak{5t-3}}$
\begin{align}
    \frac{1}{5}e^{-\frac{j6\pi f}{5}}X\brak{\frac{f}{5}} = \frac{1}{5}e^{-\frac{j6\pi f}{5}}\sinc\brak{\frac{f}{5}}
\end{align}
The correct option is (A).
