\section*{Problem}
(2.28d) Determine if $x[n] = e^{jn}$ is periodic. If it is periodic, determine its period.
\section*{Solution}
\begin{definition}
A signal $x[n]$ is said to be periodic if for all $n$, for some $N \in \mathbb{N}$, it satisfies
\begin{align}
    x[n+N] = x[n]
\end{align}
\end{definition}
The exponential signal can be generalised as,
\begin{align}
    x[n] = e^{j2\pi f n}
\end{align}
For an exponential signal to be periodic,
\begin{align}
    e^{j2\pi f(n+N)} &= e^{j2\pi f n}\\
    e^{j2\pi fN} &= 1\\
    e^{j2\pi fN} &= e^{j2\pi k}\\
    fN &= k
\end{align}
where, k is a integer.
\begin{align}
    \frac{1}{f} = \frac{N}{k}
\end{align}
Since $N$ and $k$ are integers, $\frac{N}{k} \in \mathbb{Q}$ for a periodic exponential signal. The frequency of the given signal is $f = \frac{1}{2\pi}$
\begin{align}
    \frac{N}{k} = 2\pi
\end{align}
Since $\frac{N}{k}$ is an irrational number, the given signal is aperiodic.
